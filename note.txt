
### build image
docker build -t jhondel/bulsujuans-backend-app:latest .
docker images
docker rmi bulsujuans-backend-app:latest

### run container
docker run -d -p 4000:4000 --name bulsujuans-backend-app-container --env-file .env bulsujuans-backend-app:latest
docker ps
docker rm bulsujuans-backend-app-container


### docker compose
# Start / update containers
docker compose up -d

# Stop containers
docker compose down

# Rebuild (local dev only)
docker compose up -d --build

# View logs
docker compose logs -f
